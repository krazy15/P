1. HADOOP WORDCOUNT

---

Objective:
To count frequency of words using Hadoop MapReduce.

Explanation:
MapReduce works in 3 phases:

* Map → breaks data into key-value pairs (word,1)
* Shuffle → groups same words
* Reduce → sums counts

Procedure:

# Start Hadoop services

start-dfs.sh
start-yarn.sh

# Create input file

nano input.txt

# Example text:

# big data is big data

# Create HDFS directory and upload file

hdfs dfs -mkdir /input
hdfs dfs -put input.txt /input

# Run WordCount program

hadoop jar $HADOOP_HOME/share/hadoop/mapreduce/hadoop-mapreduce-examples-*.jar wordcount /input /output

# View result

hdfs dfs -cat /output/part-r-00000

big 2
data 2

from collections import defaultdict
import re

def mapper(text):
    words= re.findall(r'\w+',text.lower())
    return[(word,1)for word in words]


def shuffle(mapped_data):
    grouped = defaultdict(list)
    for word,count in mapped_data:
        grouped[word].append(count)
    return grouped

def reducer(grouped_data):
    reduced={}
    for word,counts in grouped_data.items():
        reduced[word]= sum(counts)
    return reduced

text = "Big data is big and big data is powerful"

mapped=mapper(text)
grouped = shuffle(mapped)
result= reducer(grouped)

PRACTICAL 2: Log Analysis
from collections import defaultdict
import re

# -------- SAMPLE LOG DATA --------
logs = [
    '192.168.1.1 - - [10/Oct/2025] "GET /index.html HTTP/1.1" 404 2326',
    '192.168.1.2 - - [10/Oct/2025] "POST /login HTTP/1.1" 200 1234',
    '192.168.1.1 - - [10/Oct/2025] "GET /home HTTP/1.1" 500 532',
    '192.168.1.3 - - [10/Oct/2025] "GET /about HTTP/1.1" 404 111',
    '192.168.1.1 - - [10/Oct/2025] "GET /contact HTTP/1.1" 200 222'
]

# -------- MAP FUNCTION --------
def mapper(logs):
    mapped = []
    for line in logs:
        parts = line.split()
        
        ip = parts[0]
        status = parts[-2]
        
        mapped.append((ip, 1))
        mapped.append((f"STATUS_{status}", 1))
    
    return mapped

# -------- SHUFFLE --------
def shuffle(mapped):
    grouped = defaultdict(list)
    for key, value in mapped:
        grouped[key].append(value)
    return grouped

# -------- REDUCE --------
def reducer(grouped):
    reduced = {}
    for key, values in grouped.items():
        reduced[key] = sum(values)
    return reduced

# -------- DRIVER --------
mapped = mapper(logs)
grouped = shuffle(mapped)
result = reducer(grouped)

# Separate outputs
ip_counts = {k: v for k, v in result.items() if not k.startswith("STATUS")}
status_counts = {k: v for k, v in result.items() if k.startswith("STATUS")}

print("IP Counts:", ip_counts)
print("Status Counts:", status_counts)

PRACTICAL 11: Anonymization
data = [
    {"name": "Vikram Rawat", "email": "vikramr@gmail.com", "phone": "9876543210", "age": 21},
    {"name": "Rahul Sharma", "email": "rahul123@yahoo.com", "phone": "9123456780", "age": 35}
]

def mask_email(email):
    name,domain = email.split("@")
    return name[0]+ "****@"+ domain
    
def mask_phone(phone):
    return "******"+ phone[-4:]

def generalize_age(age):
    if age<18:
        return "Under 18"
    elif age< 30:
        return "18-29"
    elif age< 50:
        return "30-49"
    else:
        return "50+"

def anonymize_data(data):
    anonymize=[]

    for user in data:
        new_user = {
            "name":user["name"][0]+ "***",
            "email": mask_email(user["email"]),
            "phone": mask_phone(user["phone"]),
            "age": generalize_age(user["age"])
        }
        anonymize.append(new_user)

    return anonymize

result = anonymize_data(data)

for r in result:
    print(r)
